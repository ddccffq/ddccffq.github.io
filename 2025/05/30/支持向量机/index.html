\begin{align*} \min_{w, b, \xi} \quad &\frac{1}{2} \lVert w \rVert^2 + C \sum_{i = 1}^{N} \xi_i \\\\ s.t. \quad &y_i(w \cdot x_i + b) \geq 1 - \xi_i \\\\ &i = 1, 2, \ldots, N, \quad \xi_i \geq 0 \end{align*}
